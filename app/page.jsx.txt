import connectDB from '@/lib/mongodb';
import Transaction from '@/models/Transaction';
import { revalidatePath } from 'next/cache';

// Ép Next.js luôn render tươi, không render tĩnh lúc build (Sửa lỗi 404 & Prerender)
export const dynamic = 'force-dynamic';

async function addTransaction(formData) {
  'use server';
  await connectDB();
  const data = {
    title: formData.get('title'),
    amount: Number(formData.get('amount')),
    type: formData.get('type'),
    category: formData.get('category'),
    date: new Date(),
  };
  await Transaction.create(data);
  revalidatePath('/');
}

export default async function HomePage() {
  await connectDB();

  // LẤY DỮ LIỆU VÀ LÀM SẠCH (CRITICAL STEP)
  const rawTransactions = await Transaction.find()
    .sort({ date: -1 })
    .limit(10)
    .lean(); // .lean() giúp lấy object thuần, không lấy hàm của Mongoose

  const transactions = rawTransactions.map(t => ({
    ...t,
    _id: t._id.toString(), // Ép ObjectId thành String
    date: t.date ? t.date.toISOString() : new Date().toISOString(), // Ép Date thành String
  }));

  const rawTotals = await Transaction.aggregate([
    { $group: { _id: "$type", total: { $sum: "$amount" } } }
  ]);
  
  // Làm sạch dữ liệu aggregate
  const totals = JSON.parse(JSON.stringify(rawTotals));
  const income = totals.find(t => t._id === 'INCOME')?.total || 0;
  const expense = totals.find(t => t._id === 'EXPENSE')?.total || 0;

  return (
    <div className="max-w-4xl mx-auto p-4">
      <header className="mb-8 text-center">
        <h1 className="text-3xl font-bold">MyHouse 2026</h1>
        <p className="text-slate-500">Quản lý tài chính xây dựng</p>
      </header>

      <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-8 text-center">
        <div className="bg-blue-50 p-4 rounded-xl border">
          <p className="text-sm">Tổng Vốn</p>
          <p className="text-2xl font-bold text-blue-600">{income.toLocaleString()} đ</p>
        </div>
        <div className="bg-red-50 p-4 rounded-xl border">
          <p className="text-sm">Đã Chi</p>
          <p className="text-2xl font-bold text-red-600">{expense.toLocaleString()} đ</p>
        </div>
        <div className="bg-green-50 p-4 rounded-xl border">
          <p className="text-sm">Còn Lại</p>
          <p className="text-2xl font-bold text-green-600">{(income - expense).toLocaleString()} đ</p>
        </div>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        <section className="bg-white p-6 rounded-2xl shadow-sm border h-fit">
          <h2 className="text-xl font-semibold mb-4">Nhập Giao Dịch</h2>
          <form action={addTransaction} className="space-y-4">
            <input name="title" placeholder="Nội dung" required className="w-full p-2 border rounded-lg" />
            <div className="flex gap-2">
              <input name="amount" type="number" placeholder="Số tiền" required className="flex-1 p-2 border rounded-lg" />
              <select name="type" className="p-2 border rounded-lg">
                <option value="EXPENSE">Chi</option>
                <option value="INCOME">Thu</option>
              </select>
            </div>
            <button type="submit" className="w-full bg-slate-800 text-white p-3 rounded-lg font-bold">Lưu</button>
          </form>
        </section>

        <section>
          <h2 className="text-xl font-semibold mb-4">Lịch Sử</h2>
          <div className="space-y-2">
            {transactions.map((t) => (
              <div key={t._id} className="flex justify-between p-3 bg-gray-50 rounded-lg border">
                <span>{t.title}</span>
                <span className={t.type === 'EXPENSE' ? 'text-red-500 font-bold' : 'text-green-600 font-bold'}>
                  {t.type === 'EXPENSE' ? '-' : '+'}{t.amount.toLocaleString()}
                </span>
              </div>
            ))}
          </div>
        </section>
      </div>
    </div>
  );
}
