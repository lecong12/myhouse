import connectDB from '@/lib/mongodb';
import Transaction from '@/models/Transaction';
import { revalidatePath } from 'next/cache';

// CHỐT CHẶN 1: Ép Next.js không render tĩnh lúc build
export const dynamic = 'force-dynamic';

// Server Action để thêm giao dịch
async function addTransaction(formData) {
  'use server';
  await connectDB();
  const data = {
    title: formData.get('title'),
    amount: Number(formData.get('amount')),
    type: formData.get('type'),
    category: formData.get('category'),
    date: new Date(),
  };
  await Transaction.create(data);
  revalidatePath('/');
}

export default async function HomePage() {
  await connectDB();

  // CHỐT CHẶN 2: Làm sạch dữ liệu với .lean() và chuyển đổi kiểu dữ liệu
  const rawTransactions = await Transaction.find()
    .sort({ date: -1 })
    .limit(10)
    .lean(); // Trả về Plain Object thay vì Mongoose Document

  const transactions = rawTransactions.map(t => ({
    ...t,
    _id: t._id.toString(), // Bắt buộc chuyển ObjectId thành String
    date: t.date ? t.date.toISOString() : new Date().toISOString(), // Chuyển Date thành String
  }));

  // Xử lý Aggregate (nếu có) cũng cần làm sạch tương tự
  const rawTotals = await Transaction.aggregate([
    { $group: { _id: "$type", total: { $sum: "$amount" } } }
  ]);
  const totals = JSON.parse(JSON.stringify(rawTotals));

  const income = totals.find(t => t._id === 'INCOME')?.total || 0;
  const expense = totals.find(t => t._id === 'EXPENSE')?.total || 0;

  return (
    <div className="max-w-4xl mx-auto p-6">
      <h1 className="text-2xl font-bold mb-6 text-center">MyHouse Dashboard</h1>
      
      {/* Tóm tắt tài chính */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-8">
        <div className="p-4 bg-blue-50 rounded-lg border">
          <p className="text-sm">Tổng Vốn</p>
          <p className="text-xl font-bold text-blue-700">{income.toLocaleString()}đ</p>
        </div>
        <div className="p-4 bg-red-50 rounded-lg border">
          <p className="text-sm">Đã Chi</p>
          <p className="text-xl font-bold text-red-700">{expense.toLocaleString()}đ</p>
        </div>
        <div className="p-4 bg-green-50 rounded-lg border">
          <p className="text-sm">Còn Lại</p>
          <p className="text-xl font-bold text-green-700">{(income - expense).toLocaleString()}đ</p>
        </div>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        {/* Form nhập liệu */}
        <section className="bg-white p-5 rounded-xl border shadow-sm">
          <h2 className="font-semibold mb-4">Nhập giao dịch mới</h2>
          <form action={addTransaction} className="space-y-3">
            <input name="title" placeholder="Nội dung" required className="w-full p-2 border rounded" />
            <input name="amount" type="number" placeholder="Số tiền" required className="w-full p-2 border rounded" />
            <select name="type" className="w-full p-2 border rounded">
              <option value="EXPENSE">Chi</option>
              <option value="INCOME">Thu</option>
            </select>
            <button type="submit" className="w-full bg-blue-600 text-white py-2 rounded font-medium">Lưu</button>
          </form>
        </section>

        {/* Danh sách */}
        <section>
          <h2 className="font-semibold mb-4">Lịch sử gần đây</h2>
          <div className="space-y-2">
            {transactions.map(t => (
              <div key={t._id} className="flex justify-between p-3 bg-gray-50 rounded border">
                <span className="text-sm">{t.title}</span>
                <span className={`font-bold ${t.type === 'EXPENSE' ? 'text-red-500' : 'text-green-600'}`}>
                  {t.type === 'EXPENSE' ? '-' : '+'}{t.amount.toLocaleString()}
                </span>
              </div>
            ))}
          </div>
        </section>
      </div>
    </div>
  );
}
